cd (имя папки) 	- перейти в папку
cd /D			- перейти на диск D
cd C:\123		- перейти в папку 123 на диске С

dir 			- список файлов и папок в папке
CLS				- очистить окно КС

mkdir (имя папки)	- Создайте новую папку
NUL> (имя файла.расширение)		- Создайте новый файл
del (имя файла.расширение)		- Удалить файл
